Database

1. tbl_pengguna

id_pengguna INT
nis_nip_pengguna VARCHAR 18
password_pengguna VARCHAR 255
nama_pengguna VARCHAR 50
email_pengguna VARCHAR 50
alamat_pengguna VARCHAR 100
level_pengguna ENUM 
create_mdb INT
create_mdb_name VARCHAR 50
create_mdd DATETIME
mdb INT
mdb_name VARCHAR 50
mdd DATETIME


2. tbl_buku

id_buku INT
nomor_buku VARCHAR 20
kode_buku VARCHAR 20
judul_buku VARCHAR 50
pengarang_buku VARCHAR 50
penerbit_buku VARCHAR 20
tahun_terbit_buku YEAR 4
jumlah_buku INT 4
create_mdb INT
create_mdb_name VARCHAR 50
create_mdd DATETIME
mdb INT
mdb_name VARCHAR 50
mdd DATETIME

3. tbl_peminjaman

id_peminjaman INT

create_mdb INT
create_mdb_name VARCHAR 50

mdb INT

mdd DATETIME

4. tbl_pengembalian

id_pengembalian INT
